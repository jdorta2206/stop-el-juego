🧮 PUNTUACIÓN CORRECTA (CLAVE DEL STOP)
export function calculateScores(players: Record<string, PlayerAnswers>) {
  const scores: Record<string, number> = {};

  const categories = Object.keys(Object.values(players)[0]);

  for (const player in players) {
    scores[player] = 0;
  }

  for (const category of categories) {
    const answers = Object.entries(players).map(([p, a]) => ({
      player: p,
      value: normalize(a[category] || "")
    }));

    for (const { player, value } of answers) {
      if (!value) continue;

      const duplicates = answers.filter(a => a.value === value).length;

      if (duplicates === 1) scores[player] += 10;
      else scores[player] += 5;
    }
  }

  return scores;
}
